				const makeItFunny = (str, charNumber) => {
					let i = 0;
					let realIndex = 1;
					let result = '';
					while (str.length > i) {
					  if (realIndex % charNumber === 0) {
						 result = `${result}${str[i].toUpperCase()}`;
					  } else {
						 result = `${result}${str[i]}`;
					  }
					i += 1;
					realIndex+= 1;
					}
					return result;
				 };
